-- BIG Paintball - Pull + ESP (LinoriaLib)

local repo = "https://raw.githubusercontent.com/violin-suzutsuki/LinoriaLib/main/"
local Library = loadstring(game:HttpGet(repo .. "Library.lua"))()

local Window = Library:CreateWindow({
    Title = "BIG Paintball Hub",
    Center = true,
    AutoShow = true,
})

local MainTab = Window:AddTab("Main")

local pullEnabled = false
local pullRadius = 80

MainTab:AddToggle("PullToggle", { Text = "Pull Players", Default = true }):OnChanged(function(v)
    pullEnabled = v
    if v then
        task.spawn(function()
            while pullEnabled do
                local root = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
                if root then
                    for _, plr in ipairs(Players:GetPlayers()) do
                        if plr ~= LocalPlayer and plr.Character then
                            local tRoot = plr.Character:FindFirstChild("HumanoidRootPart")
                            if tRoot and (root.Position - tRoot.Position).Magnitude < pullRadius then
                                tRoot.CFrame = root.CFrame * CFrame.new(0, 4, -12)
                            end
                        end
                    end
                end
                task.wait(0.04)
            end
        end)
    end
end)

MainTab:AddSlider("Radius", { Text = "Pull Radius", Default = 80, Min = 20, Max = 150 }):OnChanged(function(v)
    pullRadius = v
end)

MainTab:AddToggle("ESP", { Text = "ESP Boxes + Tracers", Default = true })

Library:Notify("Script Loaded!", "Pull Players đã bật mặc định", 5)

LocalPlayer = Players.LocalPlayer
Players = game:GetService("Players")
