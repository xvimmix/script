local function safeLoad(url)
    local ok, result = pcall(function()
        return game:HttpGet(url)
    end)
    if not ok or type(result) ~= "string" or result:sub(1, 1) == "<" then
        error("[28th6 Hub] Khong the tai: " .. url .. "\nKiem tra mang hoac thu lai sau.", 2)
    end
    local fn, err = loadstring(result)
    if not fn then
        error("[28th6 Hub] Loi compile tu: " .. url .. "\n" .. tostring(err), 2)
    end
    return fn()
end

local function setPromptInstant(prompt, enabled)
    if not prompt or not prompt:IsA("ProximityPrompt") then
        return
    end

    if enabled then
        if State.promptHoldDurations[prompt] == nil then
            State.promptHoldDurations[prompt] = prompt.HoldDuration
        end
        prompt.HoldDuration = 0
    else
        local originalDuration = State.promptHoldDurations[prompt]
        if originalDuration ~= nil then
            prompt.HoldDuration = originalDuration
            State.promptHoldDurations[prompt] = nil
        end
    end
end

local function getPromptWorldPosition(prompt)
    if not prompt or not prompt.Parent then
        return nil
    end

    local parent = prompt.Parent

    if parent:IsA("BasePart") then
        return parent.Position
    end

    if parent:IsA("Attachment") and parent.Parent and parent.Parent:IsA("BasePart") then
        return parent.Parent.Position
    end

    local model = parent:IsA("Model") and parent or parent:FindFirstAncestorOfClass("Model")
    if model then
        local rootPart = model.PrimaryPart or model:FindFirstChild("HumanoidRootPart") or model:FindFirstChildWhichIsA("BasePart")
        if rootPart then
            return rootPart.Position
        end
    end

    return nil
end

local ProximityPromptService = game:GetService("ProximityPromptService")

local function smartTriggerPrompt(prompt)
    if not prompt or not prompt.Enabled then return end

    local originalHold = prompt.HoldDuration
    prompt.HoldDuration = 0

    local success = false

    -- Uu tien 1: InputHoldBegin/End (tu nhien nhat, it bi detect nhat)
    if not success then
        success = pcall(function()
            ProximityPromptService:InputHoldBegin(prompt, Enum.UserInputType.Keyboard)
            ProximityPromptService:InputHoldEnd(prompt, Enum.UserInputType.Keyboard)
        end)
    end

    -- Uu tien 2: Teleport gan roi ve (khong dung exploit function)
    if not success then
        local root = getRootPart()
        local promptPos = getPromptWorldPosition(prompt)

        if root and promptPos then
            local originalCFrame = root.CFrame
            local originalMaxDist = prompt.MaxActivationDistance

            prompt.MaxActivationDistance = 5
            root.CFrame = CFrame.new(promptPos + Vector3.new(0, 0, -1.5))
            task.wait()
            root.CFrame = originalCFrame

            task.delay(0.1, function()
                if prompt and prompt.Parent then
                    prompt.MaxActivationDistance = originalMaxDist
                end
            end)

            success = true
        end
    end

    -- Uu tien 3: fireproximityprompt (fallback cuoi cung)
    if not success then
        pcall(function()
            fireproximityprompt(prompt)
        end)
    end

    -- Khoi phuc HoldDuration
    task.delay(0.1, function()
        if prompt and prompt.Parent then
            prompt.HoldDuration = originalHold
        end
    end)
end

-- Dinh nghia truoc getPatchedToolState de tranh forward reference
local function restorePatchedTool(tool)
    local state = State.patchedToolStates[tool]
    if not state then
        return
    end

    for object, properties in pairs(state.values) do
        if object and object.Parent then
            for key, originalValue in pairs(properties) do
                pcall(function()
                    object[key] = originalValue
                end)
            end
        end
    end

    for moduleObject, properties in pairs(state.modules) do
        if moduleObject and moduleObject.Parent then
            pcall(function()
                local moduleValue = require(moduleObject)
                if type(moduleValue) == "table" then
                    for key, originalValue in pairs(properties) do
                        moduleValue[key] = originalValue
                    end
                end
            end)
        end
    end

    state.active.NoSpread = false
    state.active.InstantReload = false
    state.active.RapidFire = false
end
local function getPatchedToolState(tool)
    if not tool then
        return nil
    end

    local state = State.patchedToolStates[tool]
    if state then
        return state
    end

    state = {
        values = {},
        modules = {},
        connections = {},
        active = {
            NoSpread = false,
            InstantReload = false,
            RapidFire = false
        }
    }
    State.patchedToolStates[tool] = state

    state.connections.ancestryChanged = tool.AncestryChanged:Connect(function(_, parent)
        if parent ~= nil then
            return
        end

        local currentState = State.patchedToolStates[tool]
        if currentState then
            restorePatchedTool(tool)

            for _, connection in pairs(currentState.connections) do
                if connection then
                    connection:Disconnect()
                end
            end

            State.patchedToolStates[tool] = nil
        end
    end)

    return state
end

local function rememberValue(state, object, key, originalValue)
    if not object then
        return
    end

    if not state.values[object] then
        state.values[object] = {}
    end

    if state.values[object][key] == nil then
        state.values[object][key] = originalValue
    end
end

local function rememberModuleValue(state, moduleObject, key, originalValue)
    if not moduleObject then
        return
    end

    if not state.modules[moduleObject] then
        state.modules[moduleObject] = {}
    end

    if state.modules[moduleObject][key] == nil then
        state.modules[moduleObject][key] = originalValue
    end
end

local function applyToolPatchValue(state, object, key, newValue)
    if not object then
        return false
    end

    local ok, originalValue = pcall(function()
        return object[key]
    end)

    if not ok then
        return false
    end

    if type(originalValue) ~= "number" then
        return false
    end

    rememberValue(state, object, key, originalValue)

    return pcall(function()
        object[key] = newValue
    end)
end

local function applyToolPatchToModuleField(state, moduleObject, moduleValue, key, newValue)
    if type(moduleValue) ~= "table" then
        return false
    end

    local originalValue = moduleValue[key]
    if type(originalValue) ~= "number" then
        return false
    end

    rememberModuleValue(state, moduleObject, key, originalValue)
    moduleValue[key] = newValue
    return true
end

local function patchModuleNumericKeys(state, moduleObject, keyMatchers)
    if not moduleObject or not moduleObject:IsA("ModuleScript") then
        return
    end

    pcall(function()
        local moduleValue = require(moduleObject)
        if type(moduleValue) ~= "table" then
            return
        end

        for key, value in pairs(moduleValue) do
            if type(value) == "number" then
                local lowerKey = string.lower(tostring(key))
                for _, matcher in ipairs(keyMatchers) do
                    if lowerKey:find(matcher.pattern) then
                        applyToolPatchToModuleField(state, moduleObject, moduleValue, key, matcher.value)
                        break
                    end
                end
            end
        end
    end)
end


local function patchToolForInstantReload(tool)
    if not tool or not tool:IsA("Tool") then
        return
    end

    local state = getPatchedToolState(tool)
    if not state then
        return
    end

    state.active.InstantReload = true

    for _, descendant in ipairs(tool:GetDescendants()) do
        local lowerName = string.lower(descendant.Name)

        if descendant:IsA("NumberValue") or descendant:IsA("IntValue") then
            if lowerName:find("reload") or lowerName:find("cooldown") then
                applyToolPatchValue(state, descendant, "Value", 0)
            end
        elseif descendant:IsA("ModuleScript") then
            patchModuleNumericKeys(state, descendant, {
                { pattern = "reload", value = 0 },
                { pattern = "cooldown", value = 0 },
                { pattern = "reloadtime", value = 0 },
                { pattern = "reload_time", value = 0 },
                { pattern = "reloadspeed", value = 0 },
                { pattern = "reload_speed", value = 0 }
            })
        end
    end
end

local function serverHop()
    local cursor = ""
    local currentJobId = game.JobId
    local placeId = game.PlaceId

    local function buildUrl()
        local baseUrl = "https://games.roblox.com/v1/games/" .. tostring(placeId) .. "/servers/Public?sortOrder=Asc&limit=100"
        if cursor ~= "" then
            baseUrl = baseUrl .. "&cursor=" .. HttpService:UrlEncode(cursor)
        end
        return baseUrl
    end

    local ok, targetJobId = pcall(function()
        while true do
            local response = game:HttpGet(buildUrl())
            local decoded = HttpService:JSONDecode(response)

            if decoded and decoded.data then
                for _, server in ipairs(decoded.data) do
                    if server.id ~= currentJobId and server.playing < server.maxPlayers then
                        return server.id
                    end
                end
            end

            cursor = decoded and decoded.nextPageCursor or nil
            if not cursor or cursor == "" then
                break
            end
        end

        return nil
    end)

    if ok and targetJobId then
        joinJobId(targetJobId)
    elseif ok then
        notify("No available server found.", 3)
    else
        notify("Server hop failed.", 4)
    end
end

local function sendChatMessage(message)
    local text = trimString(message)
    if text == "" then
        return false
    end

    if State.textChatService and State.textChatService.ChatVersion == Enum.ChatVersion.TextChatService then
        local textChannels = State.textChatService:FindFirstChild("TextChannels")
        local generalChannel = textChannels and textChannels:FindFirstChild("RBXGeneral")
        if generalChannel and generalChannel.SendAsync then
            return pcall(function()
                generalChannel:SendAsync(text)
            end)
        end
    end

    local replicatedStorage = game:FindService("ReplicatedStorage")
    local defaultChatSystemEvents = replicatedStorage and replicatedStorage:FindFirstChild("DefaultChatSystemChatEvents")
    local sayMessageRequest = defaultChatSystemEvents and defaultChatSystemEvents:FindFirstChild("SayMessageRequest")
    if sayMessageRequest then
        return pcall(function()
            sayMessageRequest:FireServer(text, "All")
        end)
    end

    return false
end

            priority = 60,
            check = function(p)
                local va = myChar:FindFirstChild("Team") or LocalPlayer:FindFirstChild("Team")
                local vb = p.Character and (p.Character:FindFirstChild("Team") or p:FindFirstChild("Team"))
                return va and va:IsA("IntValue") and vb and vb:IsA("IntValue")
            end,
            same = function(p)
                local va = myChar:FindFirstChild("Team") or LocalPlayer:FindFirstChild("Team")
                local vb = p.Character and (p.Character:FindFirstChild("Team") or p:FindFirstChild("Team"))
                if va and vb then return va.Value == vb.Value end
                return false
            end
        },
        {
            name = "StringValue (Team)",
            priority = 60,
            check = function(p)
                local va = myChar:FindFirstChild("Team") or LocalPlayer:FindFirstChild("Team")
                local vb = p.Character and (p.Character:FindFirstChild("Team") or p:FindFirstChild("Team"))
                return va and va:IsA("StringValue") and vb and vb:IsA("StringValue") and va.Value ~= ""
            end,
            same = function(p)
                local va = myChar:FindFirstChild("Team") or LocalPlayer:FindFirstChild("Team")
                local vb = p.Character and (p.Character:FindFirstChild("Team") or p:FindFirstChild("Team"))
                if va and vb then return va.Value == vb.Value end
                return false
            end
        },
        {
            name = "Shirt Template",
            priority = 30,
            check = function(p)
                local a = myChar:FindFirstChildOfClass("Shirt")
                local b = p.Character and p.Character:FindFirstChildOfClass("Shirt")
                return a and b and a.ShirtTemplate ~= "" and b.ShirtTemplate ~= ""
            end,
            same = function(p)
                local a = myChar:FindFirstChildOfClass("Shirt")
                local b = p.Character and p.Character:FindFirstChildOfClass("Shirt")
                if a and b then return a.ShirtTemplate == b.ShirtTemplate end
                return false
            end
        },
    }

    -- Chay qua tat ca method, tim method nao:
    -- 1. Co du lieu hop le (check() = true)
    -- 2. Co kha nang phan biet: it nhat 1 nguoi CUNG doi va it nhat 1 nguoi KHAC doi
    --    Ã¢â€ â€™ gam 2 dieu nay thi method moi thuc su co ich
    -- Neu khong co du du lieu de kiem tra => fallback ve method co priority cao nhat ma check() = true
    local bestMethod = nil
    local bestScore  = -1

    for _, m in ipairs(methods) do
        local validCount   = 0
        local sameCount    = 0
        local diffCount    = 0

        for _, p in ipairs(enemies) do
            if p.Character then
                local ok = pcall(function()
                    if m.check(p) then
                        validCount = validCount + 1
                        if m.same(p) then
                            sameCount = sameCount + 1
                        else
                            diffCount = diffCount + 1
                        end
                    end
                end)
            end
        end

        if validCount > 0 then
            -- Score = priority + bonus neu phan biet duoc ca 2 phia
            local discriminationBonus = 0
            if sameCount > 0 and diffCount > 0 then
                discriminationBonus = 50 -- Method phan biet duoc ca dong doi lan ke dich
            elseif diffCount > 0 then
                discriminationBonus = 20 -- Chi phan biet ke dich (cung hop le)
            end

            local score = m.priority + discriminationBonus
            if score > bestScore then
                bestScore  = score
                bestMethod = m.name
            end
        end
    end

    return bestMethod
end

-- Team Check helper (Moved up for global access)
-- Guard: Options co the chua khoi tao neu goi qua som, dung rawget de tranh crash
local function isSameTeam(player)
    if not ESPSettings.TeamCheck then return false end

    local myChar = LocalPlayer.Character
    local theirChar = player.Character
    if not myChar or not theirChar then return false end

    -- An toan truy cap Options: neu UI chua load xong thi fallback sang Auto
    local mode = "Auto (Try All)"
    if rawget(_G, "Options") and Options and Options.TeamCheckMode then
        mode = Options.TeamCheckMode.Value or "Auto Detect"
    end
    if mode == "Auto Detect" then
        -- Chi re-scan sau moi State.teamCheckScanInterval giay (tranh spam scan moi frame)
        local now = tick()
        if now - State.lastTeamCheckScanTime > State.teamCheckScanInterval or State.detectedTeamCheckMode == nil then
            State.lastTeamCheckScanTime = now
            local detected = getAvailableTeamCheckMode()
            if detected ~= State.detectedTeamCheckMode then
                State.detectedTeamCheckMode = detected
                if detected then
                    notify("Ã°Å¸â€Â Auto detect: " .. detected, 4)
                else
                    notify("Ã¢Å¡Â Ã¯Â¸Â Auto detect: Khong tim thay method phu hop", 4)
                end
            end
            State.lastDetectedTeamCheckMode = State.detectedTeamCheckMode
        end
        mode = State.detectedTeamCheckMode or "Auto (Try All)"
    else
        State.detectedTeamCheckMode = nil
        State.lastDetectedTeamCheckMode = nil
        State.lastTeamCheckScanTime = 0
    end

    local function checkRobloxTeam()
        if LocalPlayer.Team and player.Team then
            return LocalPlayer.Team == player.Team
        end
        return false
    end

    local function checkFactionType()
        local a = myChar:GetAttribute("FactionType")
        local b = theirChar:GetAttribute("FactionType")
        if a ~= nil and b ~= nil then return a == b end
        return false
    end

    local function checkTeamAttribute()
        local a = myChar:GetAttribute("Team") or LocalPlayer:GetAttribute("Team")
        local b = theirChar:GetAttribute("Team") or player:GetAttribute("Team")
        if a ~= nil and b ~= nil then return a == b end
        return false
    end

    local function checkIntValue()
        local function getInt(char, plr)
            local v = char:FindFirstChild("Team") or plr:FindFirstChild("Team")
            if v and v:IsA("IntValue") then return v.Value end
            return nil
        end
        local a = getInt(myChar, LocalPlayer)
        local b = getInt(theirChar, player)
        if a ~= nil and b ~= nil then return a == b end
        return false
    end

    local function checkStringValue()
        local function getStr(char, plr)
            local v = char:FindFirstChild("Team") or plr:FindFirstChild("Team")
            if v and v:IsA("StringValue") then return v.Value end
            return nil
        end
        local a = getStr(myChar, LocalPlayer)
        local b = getStr(theirChar, player)
        if a ~= nil and b ~= nil and a ~= "" then return a == b end
        return false
    end

    local function checkTeamColor()
        if LocalPlayer.TeamColor and player.TeamColor then
            return LocalPlayer.TeamColor == player.TeamColor
        end
        return false
    end
    
    local function checkRoleAttribute()
        local a = myChar:GetAttribute("Role") or LocalPlayer:GetAttribute("Role")
        local b = theirChar:GetAttribute("Role") or player:GetAttribute("Role")
        if a ~= nil and b ~= nil then return a == b end
        return false
    end
    
    local function checkCrewTag()
        local a = LocalPlayer:GetAttribute("Crew") or LocalPlayer:GetAttribute("Guild")
        local b = player:GetAttribute("Crew") or player:GetAttribute("Guild")
        if a ~= nil and b ~= nil then return a == b end
        return false
    end
    
    local function checkDisplayNameColor()
        -- Game like Phantom Forces might color names in leaderboard or custom UIs, but for general purposes, if they have custom name tags in character:
        local a = myChar:FindFirstChild("Head") and myChar.Head:FindFirstChildOfClass("BillboardGui")
        local b = theirChar:FindFirstChild("Head") and theirChar.Head:FindFirstChildOfClass("BillboardGui")
        if a and b then
            local tL_A = a:FindFirstChildOfClass("TextLabel", true)
            local tL_B = b:FindFirstChildOfClass("TextLabel", true)
            if tL_A and tL_B then
                return tL_A.TextColor3 == tL_B.TextColor3
            end
        end
        return false
    end
    
    local function checkShirtTemplate()
        local a = myChar:FindFirstChildOfClass("Shirt")
        local b = theirChar:FindFirstChildOfClass("Shirt")
        if a and b and a.ShirtTemplate ~= "" and b.ShirtTemplate ~= "" then
            return a.ShirtTemplate == b.ShirtTemplate
        end
        return false
    end

    if mode == "Roblox Teams" then return checkRobloxTeam()
    elseif mode == "FactionType Attribute" then return checkFactionType()
    elseif mode == "Team Attribute" then return checkTeamAttribute()
    elseif mode == "IntValue (Team)" then return checkIntValue()
    elseif mode == "StringValue (Team)" then return checkStringValue()
    elseif mode == "TeamColor" then return checkTeamColor()
    elseif mode == "Role Attribute" then return checkRoleAttribute()
    elseif mode == "Crew / Guild Tag" then return checkCrewTag()
    elseif mode == "Display Name Color" then return checkDisplayNameColor()
    elseif mode == "Shirt Template" then return checkShirtTemplate()
    elseif mode == "Auto (Try All)" then
        if checkRobloxTeam() then return true end
        if checkFactionType() then return true end
        if checkTeamAttribute() then return true end
        if checkIntValue() then return true end
        if checkStringValue() then return true end
        if checkTeamColor() then return true end
        if checkRoleAttribute() then return true end
        if checkCrewTag() then return true end
        if checkShirtTemplate() then return true end
        -- Display Name color can be risky for try all
    end
    return false
end

local flyLV, flyAO, flyAttach = nil, nil, nil

local flyBV, flyBG = nil, nil

-- Vehicle Fly System
local vflyLV, vflyAO, vflyAttach = nil, nil, nil

local function getOccupiedVehicle()
    local char = LocalPlayer.Character
    if not char then return nil end
    local humanoid = char:FindFirstChildOfClass("Humanoid")
    if not humanoid then return nil end
    local seat = humanoid.SeatPart
    if not seat then return nil end
    -- Tim VehicleSeat hoac Seat trong Assembly
    if seat:IsA("VehicleSeat") or seat:IsA("Seat") then
        return seat
    end
    return nil
end

local function getVehicleRootPart(seat)
    if not seat then return nil end
    local model = seat:FindFirstAncestorOfClass("Model")
    if model then
        return model.PrimaryPart or seat
    end
    return seat
end

local function startVehicleFly()
    local seat = getOccupiedVehicle()
    if not seat then
        notify("Ban phai ngoi trong xe/vehicle truoc!", 3)
        State.vehicleFlyEnabled = false
        if Toggles.VehicleFlyToggle then Toggles.VehicleFlyToggle:SetValue(false) end
        return
    end
    local root = getVehicleRootPart(seat)
    if not root then return end

    -- Tat throttle/steer cua VehicleSeat de tranh xung dot
    if seat:IsA("VehicleSeat") then
        pcall(function()
            seat.MaxSpeed = 0
            seat.Throttle = 0
            seat.Steer = 0
        end)
    end

    vflyAttach = Instance.new("Attachment")
    vflyAttach.Name = "VFlyAttachment"
    vflyAttach.Parent = root

    vflyLV = Instance.new("LinearVelocity")
    vflyLV.Name = "VFlyLinearVelocity"
    vflyLV.Attachment0 = vflyAttach
    vflyLV.MaxForce = math.huge
    vflyLV.VelocityConstraintMode = Enum.VelocityConstraintMode.Vector
    vflyLV.VectorVelocity = Vector3.zero
    vflyLV.RelativeTo = Enum.ActuatorRelativeTo.World
    vflyLV.Parent = root

    vflyAO = Instance.new("AlignOrientation")
    vflyAO.Name = "VFlyAlignOrientation"
    vflyAO.Attachment0 = vflyAttach
    vflyAO.RigidityEnabled = false
    vflyAO.MaxTorque = 9e9
    vflyAO.Responsiveness = 30
    vflyAO.CFrame = root.CFrame
    vflyAO.Mode = Enum.OrientationAlignmentMode.OneAttachment
    vflyAO.Parent = root

    notify("Vehicle Fly bat! WASD/Space/Shift de dieu khien.", 3)
end

local function stopVehicleFly()
    -- Khoi phuc VehicleSeat
    local seat = getOccupiedVehicle()
    if seat and seat:IsA("VehicleSeat") then
        pcall(function()
            seat.MaxSpeed = 100
        end)
    end

    if vflyLV then vflyLV:Destroy(); vflyLV = nil end
    if vflyAO then vflyAO:Destroy(); vflyAO = nil end
    if vflyAttach then vflyAttach:Destroy(); vflyAttach = nil end
end

local function startFly()
    local root = getRootPart()
    if not root then return end
    local humanoid = getHumanoid()
    if humanoid then humanoid.PlatformStand = true end

    -- Dung Constraint-based API moi, khong bi deprecated
    flyAttach = Instance.new("Attachment")
    flyAttach.Name = "FlyAttachment"
    flyAttach.Parent = root

    flyLV = Instance.new("LinearVelocity")
    flyLV.Name = "FlyLinearVelocity"
    flyLV.Attachment0 = flyAttach
    flyLV.MaxForce = math.huge
    flyLV.VelocityConstraintMode = Enum.VelocityConstraintMode.Vector
    flyLV.VectorVelocity = Vector3.zero
    flyLV.RelativeTo = Enum.ActuatorRelativeTo.World
    flyLV.Parent = root

    flyAO = Instance.new("AlignOrientation")
    flyAO.Name = "FlyAlignOrientation"
    flyAO.Attachment0 = flyAttach
    flyAO.RigidityEnabled = false
    flyAO.MaxTorque = 9e9
    flyAO.Responsiveness = 50
    flyAO.CFrame = root.CFrame
    flyAO.Mode = Enum.OrientationAlignmentMode.OneAttachment
    flyAO.Parent = root
end

local function stopFly()
    local humanoid = getHumanoid()
    if humanoid then humanoid.PlatformStand = false end
    if flyLV then flyLV:Destroy(); flyLV = nil end
    if flyAO then flyAO:Destroy(); flyAO = nil end
    if flyAttach then flyAttach:Destroy(); flyAttach = nil end
    -- Fallback cleanup neu con object cu
    if flyBV then flyBV:Destroy(); flyBV = nil end
    if flyBG then flyBG:Destroy(); flyBG = nil end
end

local function applyLightingState()
    if State.fullbrightEnabled then
        Lighting.Ambient = Color3.new(1, 1, 1)
        Lighting.OutdoorAmbient = Color3.new(1, 1, 1)
        Lighting.Brightness = 2
        Lighting.GlobalShadows = false
    else
        Lighting.Ambient = defaultLightingState.Ambient
        Lighting.OutdoorAmbient = defaultLightingState.OutdoorAmbient
        Lighting.Brightness = defaultLightingState.Brightness
        Lighting.GlobalShadows = defaultLightingState.GlobalShadows
    end

    if State.noFogEnabled or State.fullbrightEnabled then
        Lighting.FogEnd = 1000000

        local currentAtmosphere = Lighting:FindFirstChildOfClass("Atmosphere")
        if currentAtmosphere then
            currentAtmosphere.Density = 0
            currentAtmosphere.Haze = 0
            currentAtmosphere.Glare = 0
        end

        for _, effect in ipairs(Lighting:GetChildren()) do
            if effect:IsA("BloomEffect") or effect:IsA("BlurEffect") or effect:IsA("SunRaysEffect") then
                effect.Enabled = false
            end
        end
    else
        Lighting.FogEnd = defaultLightingState.FogEnd

        local currentAtmosphere = Lighting:FindFirstChildOfClass("Atmosphere")
        if currentAtmosphere and defaultLightingState.AtmosphereDensity ~= nil then
            currentAtmosphere.Density = defaultLightingState.AtmosphereDensity
            currentAtmosphere.Haze = defaultLightingState.AtmosphereHaze
            currentAtmosphere.Glare = defaultLightingState.AtmosphereGlare
        end

        for effect, enabled in pairs(defaultLightingState.BloomEnabled) do
            if effect and effect.Parent then
                effect.Enabled = enabled
            end
        end

        for effect, enabled in pairs(defaultLightingState.BlurEnabled) do
            if effect and effect.Parent then
                effect.Enabled = enabled
            end
        end

        for effect, enabled in pairs(defaultLightingState.SunRaysEnabled) do
            if effect and effect.Parent then
                effect.Enabled = enabled
            end
        end
    end

    if State.forceTimeEnabled then
        Lighting.ClockTime = State.forceTimeValue
        -- Khoa them TimeOfDay de tranh game tu dong doi
        Lighting.TimeOfDay = string.format("%02d:00:00", math.floor(State.forceTimeValue))
    end
end

local function saveCurrentTeleportPoint()
    local root = getRootPart()
    if not root then notify("Your character root part was not found.", 3) return end
    local pointName = "Point " .. tostring(#getSavedTeleportPointNames() + 1)
    local nextIndex = 1
    while State.savedTeleportPoints[pointName] do
        nextIndex = nextIndex + 1
        pointName = "Point " .. tostring(nextIndex)
    end
    State.savedTeleportPoints[pointName] = root.CFrame
    selectedSavedTeleportPoint = pointName
    refreshSavedPointDropdown()

end

    root.CFrame = targetCFrame
end

local function refreshBangPlayerList()
    bangTargetNames = {}
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer then table.insert(bangTargetNames, player.Name) end
    end
    table.sort(bangTargetNames, function(a, b) return string.lower(a) < string.lower(b) end)
    if bangDropdown then
        bangDropdown:SetValues(bangTargetNames)
        if #bangTargetNames > 0 then
            local current = bangDropdown.Value
            if not current or current == "" or not table.find(bangTargetNames, current) then
                bangDropdown:SetValue(bangTargetNames[1])
            else
                bangTargetPlayer = current
            end
        else
            bangTargetPlayer = nil
        end
    end
end

local function isR15(character)
    return character and character:FindFirstChild("UpperTorso") ~= nil
end

local function getTorso(character)
    if not character then return nil end
    return character:FindFirstChild("HumanoidRootPart")
        or character:FindFirstChild("Torso")
        or character:FindFirstChild("UpperTorso")
end

local function unbangPlayer()
    CombatState.bangEnabled = false
    if bangState.died  then pcall(function() bangState.died:Disconnect() end);  bangState.died  = nil end
    if bangState.track then pcall(function() bangState.track:Stop() end);        bangState.track = nil end
    if bangState.anim  then pcall(function() bangState.anim:Destroy() end);      bangState.anim  = nil end
    if bangState.loop  then pcall(function() bangState.loop:Disconnect() end);    bangState.loop  = nil end
    notify("Unbang: stopped", 2)
end

local function bangSelectedPlayer()
    unbangPlayer(); task.wait()
    local myChar = getCharacter(); local myHumanoid = getHumanoid(); local myRoot = getRootPart()
    if not myChar or not myHumanoid or not myRoot then notify("Your character is not ready.", 3) return end
    if (not bangTargetPlayer or bangTargetPlayer == "") and #bangTargetNames > 0 then
        bangTargetPlayer = bangTargetNames[1]
        if bangDropdown then bangDropdown:SetValue(bangTargetPlayer) end
    end
    local targetPlayer = bangTargetPlayer and Players:FindFirstChild(bangTargetPlayer)
    local targetCharacter = targetPlayer and targetPlayer.Character
    local targetRoot = targetCharacter and targetCharacter:FindFirstChild("HumanoidRootPart")
    local targetHumanoid = targetCharacter and targetCharacter:FindFirstChildOfClass("Humanoid")
    if not targetPlayer or not targetRoot or not targetHumanoid or targetHumanoid.Health <= 0 then
        notify("Please select a valid player to bang.", 3) return
    end
    CombatState.bangEnabled = true
    bangState.anim = Instance.new("Animation")
    bangState.anim.AnimationId = isR15(myChar) and "rbxassetid://5918726674" or "rbxassetid://148840371"
    bangState.track = myHumanoid:LoadAnimation(bangState.anim)
    bangState.track:Play(0.1, 1, 1)
    bangState.track:AdjustSpeed(CombatState.bangSpeed)
    bangState.died = myHumanoid.Died:Connect(function() unbangPlayer() end)
    local bangOffset = CFrame.new(0, 0, 1.1)
    bangState.loop = RunService.Stepped:Connect(function()
        if not CombatState.bangEnabled then return end
        pcall(function()
            local otherTorso = getTorso(targetCharacter)
            local myRoot2 = getRootPart()
            if not otherTorso or not myRoot2 then unbangPlayer() return end
            myRoot2.CFrame = otherTorso.CFrame * bangOffset
        end)
    end)
    notify("Banging: " .. tostring(bangTargetPlayer), 3)
end

-- No Screen Shake (STRONG - 4 layer protection)
local noShakeConnections = {}
local SHAKE_BIND_NAME  = "StrongAntiShake"
local SHAKE_KEYWORDS   = { "shake", "screenshake", "camerashake", "shaker", "headbob", "cameraeffect", "camshake" }

local function isShakeObject(obj)
    local name = obj.Name:lower()
    for _, kw in ipairs(SHAKE_KEYWORDS) do
        if name:find(kw, 1, true) then return true end
    end
    return obj.ClassName == "ShakeController"
end

local function killShakeObject(obj)
    pcall(function()
        if obj:IsA("LocalScript") or obj:IsA("ModuleScript") then
            obj.Disabled = true
        else
            obj:Destroy()
        end
    end)
end

local function sweepForShakes(root)
    for _, obj in ipairs(root:GetDescendants()) do
        if isShakeObject(obj) then killShakeObject(obj) end
    end
end

local function startStrongAntiShake()
    -- LAYER 1: Quet va diet tat ca shake hien co
    sweepForShakes(workspace)
    sweepForShakes(Camera)
    pcall(function() sweepForShakes(game:GetService("ReplicatedStorage")) end)
    pcall(function() sweepForShakes(game:GetService("Players").LocalPlayer.PlayerGui) end)

    -- LAYER 2: Hook DescendantAdded tren workspace + Camera de block shake moi
    table.insert(noShakeConnections, workspace.DescendantAdded:Connect(function(obj)
        task.defer(function()
            if isShakeObject(obj) then killShakeObject(obj) end
        end)
    end))
    table.insert(noShakeConnections, Camera.ChildAdded:Connect(function(obj)
        task.defer(function()
            if isShakeObject(obj) or obj:IsA("BodyGyro") or obj:IsA("BodyPosition") or obj:IsA("BodyVelocity") then
                pcall(function() obj:Destroy() end)
            end
        end)
    end))

    -- LAYER 3: Block RemoteEvent co ten lien quan den shake
    pcall(function()
        for _, obj in ipairs(game:GetService("ReplicatedStorage"):GetDescendants()) do
            if obj:IsA("RemoteEvent") and isShakeObject(obj) then
                -- Wrap FireServer de no khong lam gi
                pcall(function()
                    local mt = getrawmetatable and getrawmetatable(obj)
                    if mt then
                        local old = mt.__namecall
                        mt.__namecall = function(self, ...)
                            local method = select(1, ...)
                            if self == obj and (method == "FireServer" or method == "InvokeServer") then return end
                            return old(self, ...)
                        end
                    end
                end)
            end
        end
    end)

    -- LAYER 4: BindToRenderStep tai Camera + 1 (chay SAU tat ca camera script)
    -- Su dung EMA (Exponential Moving Average) de lam pho position cao tan (shake)
    -- Alpha cao = responsive hon, alpha thap = chong shake manh hon
    -- 0.65 = can bang tot: camera van mượt, shake bi loai bỏ
    local smoothPos = Camera.CFrame.Position
    RunService:BindToRenderStep(SHAKE_BIND_NAME, Enum.RenderPriority.Camera.Value + 1, function()
        if not (Toggles.NoScreenShake and Toggles.NoScreenShake.Value) then return end
        local cf  = Camera.CFrame
        local pos = cf.Position
        -- EMA smoothing: loai bo rung dong tan so cao (shake) giu lai chuyen dong binh thuong
        smoothPos = smoothPos + (pos - smoothPos) * 0.65
        -- Giu nguyen rotation (chi smooth vi tri, khong anh huong goc nhin chuot)
        Camera.CFrame = CFrame.new(smoothPos) * (cf - pos)
    end)

    Library:Notify("No Screen Shake: ON (4 lop bao ve)", 3)
end

local function stopStrongAntiShake()
    pcall(function() RunService:UnbindFromRenderStep(SHAKE_BIND_NAME) end)
    for _, conn in ipairs(noShakeConnections) do
        pcall(function() conn:Disconnect() end)
    end
    noShakeConnections = {}
    Library:Notify("No Screen Shake: OFF", 3)
end

-- Team Check helper
local function isSameTeamForCombat(player)
    if not player or not player.Character then return false end
    local myChar = LocalPlayer.Character
    if not myChar then return false end
    -- Roblox Teams
    pcall(function()
        if LocalPlayer.Team and player.Team and LocalPlayer.Team == player.Team then
            return true
        end
    end)
    return false
end

local function isValidTarget(char)
    if not char or char == LocalPlayer.Character then return false end
    local root = char:FindFirstChild("HumanoidRootPart")
    local hum  = char:FindFirstChildOfClass("Humanoid")
    if not root or not hum or hum.Health <= 0 or hum:GetState() == Enum.HumanoidStateType.Dead then
        return false
    end
    local plr = Players:GetPlayerFromCharacter(char)
    if plr then
        if isSameTeam(plr) then return false end
    end
    if char:FindFirstChildOfClass("ForceField") then return false end
    if hum.MaxHealth >= 999999 then return false end
    local entityState = char:GetAttribute("EntityState")
    if entityState and entityState ~= 1 then return false end
    return true
end

local function isAimbotTargetVisible(targetPart)
    if not Camera or not targetPart then return false end
    local origin    = Camera.CFrame.Position
    local direction = targetPart.Position - origin
    if direction.Magnitude <= 0 then return false end
    local raycastParams = RaycastParams.new()
    raycastParams.FilterType = Enum.RaycastFilterType.Blacklist
    local filter = {}
    if LocalPlayer.Character then table.insert(filter, LocalPlayer.Character) end
    table.insert(filter, Camera)
    raycastParams.FilterDescendantsInstances = filter
    local result = Workspace:Raycast(origin, direction, raycastParams)
    return result == nil or result.Instance:IsDescendantOf(targetPart.Parent)
end

local function getTargetPartByName(char, preferredPartName)
    if not isValidTarget(char) then return nil end
    local preferredPart = char:FindFirstChild(preferredPartName)
    if preferredPart and preferredPart:IsA("BasePart") then return preferredPart end
    local fallbackParts = { "Head", "HumanoidRootPart", "UpperTorso", "Torso" }
    for _, partName in ipairs(fallbackParts) do
        local part = char:FindFirstChild(partName)
        if part and part:IsA("BasePart") then return part end
    end
    return nil
end

local function getAimbotTargetPart(char)
    return getTargetPartByName(char, CombatState.aimbotAimPart)
end


local function getBestTarget(getTargetPart, maxFov, visibleOnly)
    if not Camera then return nil end
    local mouseLocation    = UserInputService:GetMouseLocation()
    local mousePosition    = Vector2.new(mouseLocation.X, mouseLocation.Y)
    local bestCharacter    = nil
    local bestScreenDist   = maxFov

    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and player.Character and isValidTarget(player.Character) then
            local targetPart = getTargetPart(player.Character)
            if targetPart then
                local screenPoint, onScreen = Camera:WorldToViewportPoint(targetPart.Position)
                if onScreen and screenPoint.Z > 0 then
                    local screenDist = (Vector2.new(screenPoint.X, screenPoint.Y) - mousePosition).Magnitude
                    if screenDist <= bestScreenDist and (not visibleOnly or isAimbotTargetVisible(targetPart)) then
                        bestScreenDist = screenDist
                        bestCharacter  = player.Character
                    end
                end
            end
        end
    end
    return bestCharacter
end

local function getBestAimbotTarget()
    return getBestTarget(getAimbotTargetPart, CombatState.aimbotFov, CombatState.aimbotVisibleOnly)
end


-- Pull helpers
local function getPullTargetRoot(char)
    if not isValidTarget(char) then return nil end
    local root = char:FindFirstChild("HumanoidRootPart")
    if not root then return nil end
    return getRootAssembly(root) or root
end

local function getCharacterAimPoint(character)
    if not character then return nil end
    local head = character:FindFirstChild("Head")
    if head and head:IsA("BasePart") then return head.Position end
    local upperTorso = character:FindFirstChild("UpperTorso")
    if upperTorso and upperTorso:IsA("BasePart") then return upperTorso.Position + Vector3.new(0, upperTorso.Size.Y * 0.35, 0) end
    local torso = character:FindFirstChild("Torso")
    if torso and torso:IsA("BasePart") then return torso.Position + Vector3.new(0, torso.Size.Y * 0.35, 0) end
    local root = character:FindFirstChild("HumanoidRootPart")
    if root and root:IsA("BasePart") then return root.Position + Vector3.new(0, root.Size.Y * 0.5, 0) end
    return nil
end

local function getPullTargetPosition(myRoot)
    return (myRoot.CFrame * CFrame.new(
        math.round(CombatState.pullOffsetX / CombatState.pullOffsetXStep) * CombatState.pullOffsetXStep,
        math.round(CombatState.pullOffsetY / CombatState.pullOffsetYStep) * CombatState.pullOffsetYStep,
        -CombatState.pullDistance
    )).Position
end

local function getBestPullTargets(myRoot)
    local candidates = {}
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and player.Character and isValidTarget(player.Character) then
            local targetRoot     = getPullTargetRoot(player.Character)
            local targetHumanoid = player.Character:FindFirstChildOfClass("Humanoid")
            local aimPoint       = getCharacterAimPoint(player.Character)
            if targetRoot and targetHumanoid and targetHumanoid.Health > 0 and aimPoint then
                local distance = (myRoot.Position - aimPoint).Magnitude
                if distance <= CombatState.pullRadius then
                    table.insert(candidates, { character = player.Character, distance = distance, targetRoot = targetRoot })
                end
            end
        end
    end
    table.sort(candidates, function(a, b) return a.distance < b.distance end)
    local selectedTargets = {}
    for i = 1, math.min(CombatState.pullMaxTargets, #candidates) do
        selectedTargets[i] = candidates[i]
    end
    return selectedTargets
end


-- Pull loop (RenderStepped)
local cachedTargetChar = nil
RunService.RenderStepped:Connect(function()
    if not CombatState.pullEnabled or CombatState.pullPausedByChat then
        cachedTargetChar = nil
        return
    end
    local myCharacter = getCharacter()
    local myRoot      = getRootPart()
    if not myCharacter or not myRoot or shouldSuppressCharacterMotion() then
        cachedTargetChar = nil
        return
    end
    local myTool          = myCharacter:FindFirstChildOfClass("Tool")
    local selectedTargets = getBestPullTargets(myRoot)
    if #selectedTargets == 0 then cachedTargetChar = nil; return end
    cachedTargetChar = selectedTargets[1].character

    for index, targetInfo in ipairs(selectedTargets) do
        local targetRoot      = targetInfo.targetRoot
        local targetCharacter = targetInfo.character
        local targetHumanoid  = targetCharacter and targetCharacter:FindFirstChildOfClass("Humanoid")
        local targetAimPoint  = targetCharacter and getCharacterAimPoint(targetCharacter)
        if targetRoot and targetHumanoid and targetHumanoid.Health > 0 and targetAimPoint then
            local targetPosition     = getPullTargetPosition(myRoot) + Vector3.new((index - 1) * CombatState.pullLiftOffset, 0, 0)
            local currentTargetCFrame = targetRoot.CFrame
            targetRoot.CFrame = CFrame.new(targetPosition) * CFrame.Angles(currentTargetCFrame:ToOrientation())
            stopRootMotion(targetRoot)
            if myTool and myTool.Parent == myCharacter and index == 1 then
                pcall(function() myTool:Activate() end)
            end
        end
    end
end)

-- Fly loop
RunService.RenderStepped:Connect(function()
    Camera = Workspace.CurrentCamera or Camera

    if shouldSuppressCharacterMotion() then
        return
    end

    if State.flyEnabled then
        local root = getRootPart()
        if not root or not Camera then return end

        local dir = Vector3.new(0, 0, 0)
        if UserInputService:IsKeyDown(Enum.KeyCode.W) then dir = dir + Camera.CFrame.LookVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.S) then dir = dir - Camera.CFrame.LookVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.A) then dir = dir - Camera.CFrame.RightVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.D) then dir = dir + Camera.CFrame.RightVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.Space) then dir = dir + Vector3.new(0, 1, 0) end
        if UserInputService:IsKeyDown(Enum.KeyCode.LeftShift) then dir = dir - Vector3.new(0, 1, 0) end

        local velocity = dir.Magnitude > 0 and dir.Unit * State.flySpeed or Vector3.zero

        -- Dung LinearVelocity (Constraint moi)
        if flyLV then
            flyLV.VectorVelocity = velocity
        end
        if flyAO then
            flyAO.CFrame = Camera.CFrame
        end
    end

    -- Vehicle Fly loop
    if State.vehicleFlyEnabled then
        if not Camera then return end

        local seat = getOccupiedVehicle()
        if not seat then
            -- Xe bi roi khi dang bay Ã¢â€ â€™ tat vehicle fly
            stopVehicleFly()
            State.vehicleFlyEnabled = false
            if Toggles.VehicleFlyToggle then Toggles.VehicleFlyToggle:SetValue(false) end
        else
            local vRoot = getVehicleRootPart(seat)
            if vRoot then
                local dir = Vector3.zero
                if UserInputService:IsKeyDown(Enum.KeyCode.W) then dir = dir + Camera.CFrame.LookVector end
                if UserInputService:IsKeyDown(Enum.KeyCode.S) then dir = dir - Camera.CFrame.LookVector end
                if UserInputService:IsKeyDown(Enum.KeyCode.A) then dir = dir - Camera.CFrame.RightVector end
                if UserInputService:IsKeyDown(Enum.KeyCode.D) then dir = dir + Camera.CFrame.RightVector end
                if UserInputService:IsKeyDown(Enum.KeyCode.Space) then dir = dir + Vector3.new(0, 1, 0) end
                if UserInputService:IsKeyDown(Enum.KeyCode.LeftShift) then dir = dir - Vector3.new(0, 1, 0) end

                -- Loai bo component Y khi k an Space/Shift de xe khong bi keo xuong
                local speed = State.vehicleFlySpeed
                local velocity = dir.Magnitude > 0 and dir.Unit * speed or Vector3.zero

                if vflyLV then
                    vflyLV.VectorVelocity = velocity
                end
                if vflyAO and dir.Magnitude > 0 then
                    -- Giup xe quay mat theo huong di chuyen
                    vflyAO.CFrame = CFrame.new(Vector3.zero, dir.Unit)
                end
            end
        end
    end
end)

RunService.Heartbeat:Connect(function()
    Camera = Workspace.CurrentCamera or Camera

    local character = LocalPlayer.Character
    if character then
        applyCharacterFootState(character)
    end

    if shouldSuppressCharacterMotion() then
        return
    end

    local root = getRootPart()
    local humanoid = getHumanoid()
    local equippedTool = character and character:FindFirstChildOfClass("Tool")

    if State.spamChatEnabled then
        local now = tick()
        local message = trimString(State.spamChatMessage)
        if message ~= "" and now - State.lastSpamChatTime >= State.spamChatDelay then
            if sendChatMessage(message) then
                State.lastSpamChatTime = now
            end
        end
    end

    if State.autoPickupEnabled then
        local now = tick()
        if now - State.lastAutoPickupTime > 0.3 then
            State.lastAutoPickupTime = now
            task.spawn(autoPickupNearbyItems)
        end
    end

    for tool, state in pairs(State.patchedToolStates) do
        state.active.NoSpread = false
        state.active.InstantReload = false
        state.active.RapidFire = false

        local shouldKeepPatched = tool == equippedTool and (
            State.noSpreadEnabled or State.instantReloadEnabled or State.rapidFireEnabled
        )

        if not shouldKeepPatched then
            restorePatchedTool(tool)
        end
    end

    if equippedTool then
        if State.noSpreadEnabled then
            patchToolForNoSpread(equippedTool)
        end

        if State.instantReloadEnabled then
            patchToolForInstantReload(equippedTool)
        end

        -- Blaster Instant Reload: fire Reload remote cho he thong Blaster (sung 1 vien nhu RPG)
        if State.instantReloadEnabled then
            local blaster = game:GetService("ReplicatedStorage"):FindFirstChild("Blaster")
            if blaster then
                local remotes = blaster:FindFirstChild("Remotes")
                local reloadRemote = remotes and remotes:FindFirstChild("Reload")
                if reloadRemote then
                    -- Tim vu khi trong Character (RPG, M4A1, v.v.)
                    local blasterWeapon = equippedTool
                    if not blasterWeapon then
                        for _, child in ipairs(LocalPlayer.Character:GetChildren()) do
                            if (child:IsA("Tool") or child:IsA("Model")) and not child:IsA("Accessory") then
                                if child.Name ~= "Head" and child.Name ~= "Torso" and child.Name ~= "HumanoidRootPart" and not string.match(child.Name, "Arm") and not string.match(child.Name, "Leg") then
                                    blasterWeapon = child
                                    break
                                end
                            end
                        end
                    end
                    if blasterWeapon then
                        pcall(function() reloadRemote:FireServer(blasterWeapon) end)
                    end
                end
            end
        end

        if State.rapidFireEnabled then
            patchToolForRapidFire(equippedTool)
        end
    end

    if humanoid then
        if State.speedEnabled then
            local method = Options.SpeedMethod and Options.SpeedMethod.Value or "WalkSpeed"
            local moving = humanoid.MoveDirection.Magnitude > 0

            if method == "WalkSpeed" then
                humanoid.WalkSpeed = State.walkSpeed
            elseif method == "CFrame" and moving and root then
                root.CFrame = root.CFrame + (humanoid.MoveDirection * (State.walkSpeed / 60))
            elseif method == "Velocity" and moving and root then
                local horizontalVelocity = humanoid.MoveDirection * State.walkSpeed
                root.AssemblyLinearVelocity = Vector3.new(horizontalVelocity.X, root.AssemblyLinearVelocity.Y, horizontalVelocity.Z)
            end
        else
            local defaultWalkSpeed = 16
            if humanoid.WalkSpeed ~= defaultWalkSpeed then
                humanoid.WalkSpeed = defaultWalkSpeed
            end
        end
    end
    
    if State.noclipEnabled then
        local char = getCharacter()
        if char then
            for _, part in pairs(char:GetDescendants()) do

local function createSkeletonLine()
    local line = Drawing.new("Line")
    line.Color = ESPSettings.EnemyColor
    line.Thickness = 1.5
    line.Visible = false
    line.ZIndex = 2
    return line
end

local function getSkeletonPoint(char, partName)
    local part = char and char:FindFirstChild(partName)
    if not part or not part:IsA("BasePart") then
        return nil
    end

    local position, onScreen = Camera:WorldToViewportPoint(part.Position)
    if not onScreen then
        return nil
    end

    return Vector2.new(position.X, position.Y)
end

local function setSkeletonLine(line, fromPoint, toPoint)
    if line and fromPoint and toPoint then
        line.From = fromPoint
        line.To = toPoint
        line.Visible = true
    elseif line then
        line.Visible = false
    end
end

local function hideSkeletonLines(skeletonLines)
    if not skeletonLines then
        return
    end

    for _, line in pairs(skeletonLines) do
        if line then
            line.Visible = false
        end
    end
end

local function getCharacterWeaponName(char)
    if not char then
        return nil
    end

    local tool = char:FindFirstChildOfClass("Tool")
    if tool and tool.Name and tool.Name ~= "" then
        return tool.Name
    end

    return nil
end

local function updateSkeletonESP(char, esp)
    if not esp or not esp.SkeletonLines then
        return
    end

    if not ESPSettings.Skeleton then
        hideSkeletonLines(esp.SkeletonLines)
        return
    end

    local myRoot = getRootPart()
    local targetRoot = char and char:FindFirstChild("HumanoidRootPart")
    if myRoot and targetRoot then
        local distance = (myRoot.Position - targetRoot.Position).Magnitude
        if distance > 250 then
            hideSkeletonLines(esp.SkeletonLines)
            return
        end
    end

    local head = getSkeletonPoint(char, "Head")
    local upperTorso = getSkeletonPoint(char, "UpperTorso") or getSkeletonPoint(char, "Torso")
    local lowerTorso = getSkeletonPoint(char, "LowerTorso") or upperTorso
    local leftUpperArm = getSkeletonPoint(char, "LeftUpperArm") or getSkeletonPoint(char, "Left Arm")
    local leftLowerArm = getSkeletonPoint(char, "LeftLowerArm")
    local leftHand = getSkeletonPoint(char, "LeftHand")
    local rightUpperArm = getSkeletonPoint(char, "RightUpperArm") or getSkeletonPoint(char, "Right Arm")
    local rightLowerArm = getSkeletonPoint(char, "RightLowerArm")
    local rightHand = getSkeletonPoint(char, "RightHand")
    local leftUpperLeg = getSkeletonPoint(char, "LeftUpperLeg") or getSkeletonPoint(char, "Left Leg")
    local leftLowerLeg = getSkeletonPoint(char, "LeftLowerLeg")
    local leftFoot = getSkeletonPoint(char, "LeftFoot")
    local rightUpperLeg = getSkeletonPoint(char, "RightUpperLeg") or getSkeletonPoint(char, "Right Leg")
    local rightLowerLeg = getSkeletonPoint(char, "RightLowerLeg")
    local rightFoot = getSkeletonPoint(char, "RightFoot")

    if not head or not upperTorso then
        hideSkeletonLines(esp.SkeletonLines)
        return
    end

    local neckPoint = Vector2.new((head.X + upperTorso.X) / 2, (head.Y + upperTorso.Y) / 2)

    setSkeletonLine(esp.SkeletonLines.HeadToTorso, head, neckPoint)
    setSkeletonLine(esp.SkeletonLines.UpperToLowerTorso, upperTorso, lowerTorso)

    setSkeletonLine(esp.SkeletonLines.LeftUpperArm, upperTorso, leftUpperArm)
    setSkeletonLine(esp.SkeletonLines.LeftLowerArm, leftUpperArm, leftLowerArm or leftHand)
    setSkeletonLine(esp.SkeletonLines.LeftHand, leftLowerArm, leftHand)

    setSkeletonLine(esp.SkeletonLines.RightUpperArm, upperTorso, rightUpperArm)
    setSkeletonLine(esp.SkeletonLines.RightLowerArm, rightUpperArm, rightLowerArm or rightHand)
    setSkeletonLine(esp.SkeletonLines.RightHand, rightLowerArm, rightHand)

    setSkeletonLine(esp.SkeletonLines.LeftUpperLeg, lowerTorso, leftUpperLeg)
    setSkeletonLine(esp.SkeletonLines.LeftLowerLeg, leftUpperLeg, leftLowerLeg or leftFoot)
    setSkeletonLine(esp.SkeletonLines.LeftFoot, leftLowerLeg, leftFoot)

    setSkeletonLine(esp.SkeletonLines.RightUpperLeg, lowerTorso, rightUpperLeg)
    setSkeletonLine(esp.SkeletonLines.RightLowerLeg, rightUpperLeg, rightLowerLeg or rightFoot)
    setSkeletonLine(esp.SkeletonLines.RightFoot, rightLowerLeg, rightFoot)
end

local function RemoveESP(char)
    if ESP_Cache[char] then
        for k, v in pairs(ESP_Cache[char]) do
            if k == "Highlight" and v then
                v:Destroy()
            elseif k == "SkeletonLines" and v then
                for _, line in pairs(v) do
                    pcall(function()
                        line.Visible = false
                        line:Remove()
                    end)
                end
            else
                pcall(function()
                    v.Visible = false
                    v:Remove()
                end)
            end
        end
        ESP_Cache[char] = nil
    end
end

                            esp.Distance.Text = "[" .. tostring(dist) .. "m]"
                            esp.Distance.Position = Vector2.new(rootPos.X, rootPos.Y + height / 2 + 2)
                            esp.Distance.Visible = true
                        else
                            esp.Distance.Visible = false
                        end

                        if ESPSettings.Weapon then
                            local weaponName = getCharacterWeaponName(char)
                            if weaponName then
                                esp.Weapon.Text = weaponName
                                esp.Weapon.Position = Vector2.new(rootPos.X, rootPos.Y + height / 2 + (ESPSettings.Distance and 18 or 2))
                                esp.Weapon.Visible = true
                            else
                                esp.Weapon.Visible = false
                            end
                        else
                            esp.Weapon.Visible = false
                        end
                        
                        if ESPSettings.Tracers then
                            esp.Tracer.From = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y)
                            esp.Tracer.To = Vector2.new(rootPos.X, rootPos.Y + height / 2)
                            esp.Tracer.Visible = true
                        else
                            esp.Tracer.Visible = false
                        end

                        updateSkeletonESP(char, esp)
                    else
                        esp.Box.Visible = false; esp.BoxOutline.Visible = false
                        esp.HealthBar.Visible = false; esp.HealthBarOutline.Visible = false
                        esp.Name.Visible = false; esp.Distance.Visible = false; esp.Weapon.Visible = false; esp.Tracer.Visible = false
                        hideSkeletonLines(esp.SkeletonLines)
                    end
                    
                    if ESPSettings.Chams then
                        if not esp.Highlight or esp.Highlight.Parent ~= char then
                            if esp.Highlight then esp.Highlight:Destroy() end
                            local hl = Instance.new("Highlight")
                            hl.Name = "ESPCham"
                            hl.FillTransparency = 0.5
                            hl.OutlineTransparency = 0
                            hl.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
                            hl.Parent = char
                            esp.Highlight = hl
                        end
                        esp.Highlight.FillColor = ESPSettings.EnemyColor
                        esp.Highlight.OutlineColor = Color3.new(1, 1, 1)
                    else
                        if esp.Highlight then esp.Highlight:Destroy(); esp.Highlight = nil end
                    end
                    
                else
                    esp.Box.Visible = false; esp.BoxOutline.Visible = false
                    esp.HealthBar.Visible = false; esp.HealthBarOutline.Visible = false
                    esp.Name.Visible = false; esp.Distance.Visible = false; esp.Weapon.Visible = false; esp.Tracer.Visible = false
                    hideSkeletonLines(esp.SkeletonLines)
                    if esp.Highlight then esp.Highlight:Destroy(); esp.Highlight = nil end
                end
            end
        end
    end
end)

RunService:BindToRenderStep("AimbotCamOverride", Enum.RenderPriority.Camera.Value + 1, function()
    if not CombatState.aimbotEnabled or not Camera then return end
    if not CombatState.aimbotTarget then return end

    local shouldAim = true
    if CombatState.aimbotHoldEnabled then
        shouldAim = UserInputService:IsMouseButtonPressed(Enum.UserInputType.MouseButton2)
    end
    if not shouldAim then return end

    local targetPart = getAimbotTargetPart(CombatState.aimbotTarget)
    if not targetPart or not isValidTarget(CombatState.aimbotTarget) then
        CombatState.aimbotTarget = nil
        return
    end
    if CombatState.aimbotVisibleOnly and not isAimbotTargetVisible(targetPart) then return end

    local origin = Camera.CFrame.Position
    local destination = targetPart.Position
    local aimCFrame = CFrame.new(origin, destination)
    Camera.CFrame = Camera.CFrame:Lerp(aimCFrame, math.clamp(CombatState.aimbotSmoothness, 0.01, 1))
end)

-- =====================================================================
-- Memory cleanup khi script unload / game dong
-- =====================================================================
local function fullCleanup()
    -- Disconnect tat ca tracked connections
    for _, conn in ipairs(Connections) do
        pcall(function() conn:Disconnect() end)
    end
    Connections = {}

    -- Xoa tat ca ESP Drawing objects chua duoc cleanup
    for char, _ in pairs(ESP_Cache) do
        pcall(function() RemoveESP(char) end)
    end

    -- Cleanup State instances
    if State.floatBodyVelocity then
        pcall(function() State.floatBodyVelocity:Destroy() end)
        State.floatBodyVelocity = nil
    end
    if State.flingBav then
        pcall(function() State.flingBav:Destroy() end)
        State.flingBav = nil
    end

    -- Cleanup Drawing objects cho aimbot FOV circle
    pcall(function()
        aimbotFovCircle.Visible = false
        aimbotFovCircle:Remove()
    end)
    pcall(function()
        if aimbotTargetText then
            aimbotTargetText.Visible = false
            aimbotTargetText:Remove()
        end
    end)
end

-- Chay cleanup khi player bi kick / teleport
LocalPlayer.AncestryChanged:Connect(function(_, parent)
    if not parent then
        fullCleanup()
    end
end)
